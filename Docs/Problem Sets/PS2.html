def equality(value, total_value):
    for i in range(1, 51):
        top = np.percentile(value, 100-i)
        total = value[value > top].sum()
        if ((total/total_value) > (100-i)/100): 
            i = i - 1
        if ((total/total_value) < (100-i)/100): 
            i = i + 1
        else:
            value = i
            break
    return(value)
